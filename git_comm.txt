git config --global init.defaultBranch main  установить имя main для вашей ветки по умолчанию

git config --list проверить используемую конфигурацию

git status - для определения, какие файлы в каком состоянии находятся
git status -s(git status --short) - упрощённый вывод
  ?? -  неотслеживаемые файлы
  A  -  добавленные в отслеживаемые
  M  - отредактированные файлы

файл .gitignore - игнорирование файлов
   Символ (*) соответствует 0 или более символам
    [abc] — любому символу из указанных в скобках
    [0-9] — соответствуют любому символу из интервала
    a/**/z - использовать две звёздочки, чтобы указать на вложенные каталоги: a/**/z соответствует a/z, a/b/z, a/b/c/z
    # Исключить все файлы с расширением .a
    *.a
    # Но отслеживать файл lib.a даже если он подпадает под исключение выше
    !lib.a
    # Исключить файл TODO в корневом каталоге, но не файл в subdir/TODO
    /TODO
    # Игнорировать все файлы в каталоге build/
    build/
    # Игнорировать файл doc/notes.txt, но не файл doc/server/arch.txt
    doc/*.txt
    # Игнорировать все .txt файлы в каталоге doc/
    doc/**/*.txt

git diff - непосредственно добавленные и удалённые строки — патч как он есть(только те, что ещё не проиндексированы)
git diff --staged(git diff --cached) -равнивает ваши проиндексированные изменения с последним коммитом

$ git commit - зафиксировать изменения   
git commit -m "комментарий к изменению"
